## FILE 7: Makefile

.PHONY: help install uninstall test clean build

VERSION := 4.0.0
INSTALL_DIR := $(HOME)/.universal-workspace

help:
	@echo "Universal Workspace Platform - Makefile"
	@echo ""
	@echo "Targets:"
	@echo "  install      Install UWP"
	@echo "  uninstall    Uninstall UWP"
	@echo "  reinstall    Reinstall UWP"
	@echo "  test         Run tests"
	@echo "  build        Build distribution"
	@echo "  clean        Clean build artifacts"
	@echo "  docker-build Build Docker image"
	@echo "  docker-run   Run Docker container"

install:
	@bash install.sh --no-gui

uninstall:
	@bash uninstall.sh --force

reinstall: uninstall install

test:
	@bash tests/test-suite.sh

build:
	@bash scripts/deploy.sh build

clean:
	@rm -rf dist/ build/ *.tar.gz
	@echo "Clean complete"

docker-build:
	@docker build -t uwp:latest .

docker-run:
	@docker run -it uwp:latest

## FILE 9: CONTRIBUTING.md

# Contributing to Universal Workspace Platform

## Code of Conduct

This project adheres to a Contributor Code of Conduct.

## Getting Started

1. Fork the repository
2. Clone your fork
3. Create a feature branch
4. Make your changes
5. Test your changes
6. Submit a pull request

## Development Setup

```bash
git clone https://github.com/username/uwp.git
cd uwp
bash install.sh --no-gui
```

## Testing

```bash
# Run test suite
bash tests/test-suite.sh

# Run specific test
bash tests/unit-tests/test-core.sh

# Run with coverage
bash tests/test-suite.sh --coverage
```

## Code Style

- Bash scripts: Follow Google Shell Style Guide
- Comments: Clear and descriptive
- Functions: Use uwp_ prefix
- Variables: Use UPPER_CASE for constants
- Error handling: Use proper error messages

## Commit Messages

Format: `type(scope): subject`

Types:
- feat: New feature
- fix: Bug fix
- docs: Documentation
- test: Tests
- chore: Build/tooling

Example:
```
feat(ai): Add code suggestion feature
fix(android): Fix ADB device detection
docs(readme): Update installation instructions
```

## Pull Request Process

1. Update documentation
2. Run tests
3. Update CHANGELOG
4. Create PR with description
5. Address review feedback
6. Wait for approval
7. Merge

## Reporting Issues

- Use GitHub Issues
- Include system info: `uwp diagnose`
- Include logs and errors
- Steps to reproduce
- Expected vs actual

## Feature Requests

- Discuss in GitHub Discussions
- Check for duplicates
- Provide use cases
- Be open to feedback

## FILE 13: .github/workflows/ci.yml

name: CI

on:
  push:
    branches: [ main, develop ]
  pull_request:
    branches: [ main, develop ]

jobs:
  test:
    runs-on: ubuntu-latest
    strategy:
      matrix:
        bash: [4.0, 5.0, 5.1]
    steps:
      - uses: actions/checkout@v3
      - name: Run tests
        run: |
          bash tests/test-suite.sh
          bash scripts/validate.sh

  lint:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v3
      - name: Lint shell scripts
        run: |
          find . -name "*.sh" -type f | xargs shellcheck -x

  security:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v3
      - name: Security scan
        run: |
          bash tests/validators/security-check.sh

## FILE 14: .github/workflows/release.yml

name: Release

on:
  push:
    tags:
      - 'v*'

jobs:
  release:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v3
      - name: Build distribution
        run: bash scripts/deploy.sh build
      - name: Create release
        uses: softprops/action-gh-release@v1
        with:
          files: dist/*
          draft: false
          prerelease: false
